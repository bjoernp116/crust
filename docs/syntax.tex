\section*{Grammar}

\begin{align}
    [\text{Program}] &\to [\text{Stmt}]^* \\
    [\text{Stmt}] &\to 
        \begin{cases}
            [\text{exit([Expr])}] \\
            [\text{VarDecl}] \\
        \end{cases} \\
    \text{[VarDecl]} &\to \text{let [Identifier] = [Expr]} \\
    [\text{Expr}] &\to 
        \begin{cases}
            [\text{Term}] \\
            [\text{BinExpr}] \\
        \end{cases} \\
    [\text{BinExpr}] &\to 
        \begin{cases}
            [\text{Expr}] * [\text{Expr}] \\
            [\text{Expr}] / [\text{Expr}] \\
            [\text{Expr}] + [\text{Expr}] \\
            [\text{Expr}] - [\text{Expr}] \\
        \end{cases} \\
    [\text{Term}] &\to 
        \begin{cases}
            [\text{Identifier}] \\
            [\text{IntLitteral}] \\
            ([\text{Expr}]) \\
        \end{cases} \\
    [\text{Identifier}] &\to (a..z \cup A..Z \cup 0..9)^* \\
    [\text{IntLitteral}] &\to (0..9)^* \\
\end{align}
